git init	# create a git
git config --global user.email test@gmail.com	# set email
git config --global user.name Davood	#set name
git status	# check status
git add test.txt	# add a file to stage
git add -A	# add all file to stage
git commit -m "content of that commit"	# commit all file that exist in stage
git reset test.txt	# comeback a file from stage
git restore	# after a file went to stage this command comeback a file to first mode 
git log		#commit logs
git checkout 8d3fe409d42d50afa1b0995eff64d25d8f4933e1 -- test.txt	# reset a file to a commit 
git checkout -- test	# back to last commit before change
git checkout branch_name	# change branch
git branch	# show all branch
git branch branch_name	# create a new branch
git checkout branch_name	# switch to other branch
git switch branch_name	# switch to other branch 
git merge branch_name	# merge another branch to master branch 
git rm test.txt		# delete a file
